cmake_minimum_required(VERSION 3.6)
project(StabilizerCPP)

set(CMAKE_CXX_STANDARD 11)

MESSAGE(${CMAKE_SOURCE_DIR})
set(NO_BOOST_CMAKE TRUE)
find_package(BOOST REQUIRED)
if(Boost_FOUND)
    include_directories(${Boost_INCLUDE_DIRS})
endif()
include_directories( ${CMAKE_SOURCE_DIR}/src/lib )
set(SOURCE_FILES src/main.cpp src/SymplecticPauli.cpp src/StabilizerGroup.cpp src/utils.cpp)
add_executable(StabilizerCPP ${SOURCE_FILES})
target_link_libraries(StabilizerCPP ${Boost_LIBRARIES})
